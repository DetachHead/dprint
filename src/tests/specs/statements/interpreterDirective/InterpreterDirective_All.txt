#!/usr/bin/env node
1+2;

[expect]
#!/usr/bin/env node
1 + 2;

#!/usr/bin/env node

1+2;

#!/usr/bin/env node

1 + 2;

#!/usr/bin/env node


1+2;

[expect]

1 + 2;
